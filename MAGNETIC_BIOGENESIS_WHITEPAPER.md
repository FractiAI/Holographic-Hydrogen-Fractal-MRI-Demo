async function retrieveSeed(walletAddress: string, seedVersion: number) {
  // Query Base L2 blockchain for NFT
  const nft = await blockchain.getNFT(walletAddress, seedVersion)
  
  // Verify authenticity
  const valid = await cryptography.verify(nft.signature)
  if (!valid) throw new Error("Invalid seed signature")
  
  // Retrieve seed data from IPFS
  const ipfsHash = nft.metadata.seedDataURI
  const encryptedSeed = await ipfs.get(ipfsHash)
