        question_grammar['curiosity_vs_extraction'] * 0.25 +
        exploration_fractality * 0.20 +
        completion_motivation['intrinsic_vs_extrinsic'] * 0.20 +
        synchronicity_score * 0.10
    )

    if resonance_score >= 0.87:
        return True, resonance_score, "RESONANT_ONE"
    elif resonance_score >= 0.70:
        return False, resonance_score, "CURIOUS_BUT_INCOMPLETE"
    else:
        return False, resonance_score, "MATERIAL_MOTIVATED"

def calculate_depth_vs_breadth(session):
    """Users who linger on complex sections vs skip for highlights."""
    time_on_equations = sum(session.time_spent[s] for s in session.sections if 'equation' in s.type)
    time_on_summaries = sum(session.time_spent[s] for s in session.sections if 'summary' in s.type)
    return time_on_equations / (time_on_summaries + 1e-6)

def analyze_question_semantics(questions):
    """Curiosity-driven questions differ grammatically from extraction-focused."""
    curiosity_markers = ['how does', 'why would', 'what happens when', 'can you explain']
    extraction_markers = ['how much', 'what\'s the roi', 'can I sell', 'market size']
    
    curiosity_score = sum(1 for q in questions if any(m in q.lower() for m in curiosity_markers))
    extraction_score = sum(1 for q in questions if any(m in q.lower() for m in extraction_markers))
    
    return curiosity_score / (curiosity_score + extraction_score + 1e-6)

def measure_self_similar_curiosity(exploration_path):
    """Genuine curiosity exhibits fractal patterns (deep dives repeat across topics)."""
    # If user explores hydrogen deeply, AND explores SSAN deeply, AND explores biogenesis deeply
    # (same *style* of exploration repeated) = fractal signature
    depth_per_section = [len(path) for path in exploration_path.grouped_by_section()]
    variance = np.var(depth_per_section)
    return 1.0 / (1.0 + variance)  # Low variance = consistent deep exploration = fractal

def infer_why_completed(timing, return_visits):
    """Did they complete because they had to, or because they wanted to?"""
    if len(return_visits) > 3:  # Came back multiple times
        return 0.95  # Strong intrinsic
    if timing['total_duration'] > 2 * timing['minimum_required']:
        return 0.90  # Spent way more time than needed
    if timing['completed_in_one_session']:
        return 0.70  # Could be intrinsic OR driven
    return 0.50  # Ambiguous

def detect_impossible_knowledge(comments):
    """Sometimes resonant ones reference concepts before they appear in the demo."""
    # This is rare but notable: Synchronicity events
    # E.g., mentioning "fractal temporal compression" before reaching that section
    # Suggests: Either they read ahead (unlikely with our layout), or... resonance
    return analyze_for_acausal_knowledge(comments)
